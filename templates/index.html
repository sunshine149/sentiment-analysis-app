cURL命令示例

  -H "Content-Type: application/json" \
  -d '{"text": "这家酒店真不错"}'

# 提交反馈

  -H "Content-Type: application/json" \

# 开始训练（指定模型类型）
curl -X POST http://localhost:5000/api/train \
  -H "Content-Type: application/json" \
  -d '{"use_gpu": false, "model_type": "random_forest"}'

# 获取训练历史
curl -X GET http://localhost:5000/api/training_history
